#include <stdio.h>
#include <math.h>
#define MAX 30
#define MAXB 15
void main()
{
  int i,j,tb,n,m,k;
  float x[MAX],y[MAX],p[MAX],pmax,sum,bx[MAXB],by[MAXB],d2;
  printf("Δώστε τον αριθμό των αναμεταδοτών (μέγιστο %d) = ? ",MAX);
  scanf("%d",&n);
  printf("Δώστε τον αριθμό των σταθμών βάσης (μέγιστο %d) = ? ",MAXB);
  scanf("%d",&m);
  
  printf("Δώστε τις συντεταγμένες και την ισχύ του κάθε αναμεταδότη\n");
  for(i=0;i<n;i++){
    printf("Αναμεταδότης %d\n",i);
    scanf("%f %f",&x[i],&y[i]);
    scanf("%f",&p[i]);
  }
  printf("Δώστε τις συντεταγμένες των σταθμών βάσης\n");
  for(i=0;i<m;i++){
    printf("Σταθμός βάσης %d\n",i);
    scanf("%f %f",&bx[i],&by[i]);
  }
  for(j=0;j<m;j++){
    sum=0;
    k=0;
    for(i=0;i<n;i++){
      d2=pow(bx[j]-x[i],2)+pow(by[j]-y[i],2);
      if(d2==0){
	printf("Ο αναμεταδότης %d δε λαμβάνεται υπ όψιν για τη θέση %d\n",i,j);
	continue;
      }
      k++;
      sum+=p[i]/d2;
    }
    sum/=k;
    if(j==0){
      pmax=sum;
      tb=j;
      continue;
    }
    if(sum>pmax){
      tb=j;
      pmax=sum;
    }
  } 
  printf("Ο σταθμός πρέπει να εγκατασταθεί στη θέση με συντεταγμένες %f %f\n",bx[tb],by[tb]);
}
