#include <stdio.h>
#include <alloc.h>
#include <process.h>
#include <math.h>
void calk(int k,float *x,float *y,float *p,int n,float minp,float *mean,int *m);
void main()
{
  float *x,*y,*p,minp,mean,maxmean;
  int i,k,n,m,max_i,m_max;
  printf("Δώστε τον αριθμό των πυλώνων = ? ");
  scanf("%d",&n);
  printf("Δώστε την ελάχιστη αποδεκτή τιμή για την ισχύ του σήματος = ? ");
  scanf("%f",&minp);
  if((x=(float *)malloc(n*sizeof(float)))==NULL){
    printf("Δεν υπάρχει αρκετή διαθέσιμη μνήμη στη θέση 1\n");
    exit(1);
  }
  if((y=(float *)malloc(n*sizeof(float)))==NULL){
    printf("Δεν υπάρχει αρκετή διαθέσιμη μνήμη στη θέση 2\n");
    exit(2);
  }
  if((p=(float *)malloc(n*sizeof(float)))==NULL){
    printf("Δεν υπάρχει αρκετή διαθέσιμη μνήμη στη θέση 3\n");
    exit(3);
  }
  for(i=0;i<n;i++){
    printf("Δώστε τις συντεταγμένες x,y για τον πυλώνα %d = ? ",i+1);
    scanf("%f %f",&x[i],&y[i]);
    printf("Δώστε την ισχύ με την οποία εκπέμπει ο πυλώνας %d = ? ",i+1);
    scanf("%f",&p[i]);
  }
  calk(0,x,y,p,n,minp,&maxmean,&m);
  max_i=0;
  m_max=m;
  for(i=1;i<n;i++){
    calk(i,x,y,p,n,minp,&mean,&m);
    if(m_max<m){
      maxmean=mean;
      m_max=m;
      max_i=i;
      continue;
    }
    if(m_max==m)
      if(maxmean<mean){
        maxmean=mean;
	max_i=i;
      }
  }
  printf("\nΟ σταθμός βάσης πρέπει να εγκατασταθεί στη θέση του πυλώνα %d\n",max_i+1);
  free(x);
  free(y);
  free(p);
}

void calk(int k,float *x,float *y,float *p,int n,float minp,float *mean,int *m)
{
  int i;
  float d,w;
  *m=0;
  *mean=0;
  for(i=0;i<n;i++){
    if(i==k)continue;
    d=pow(x[k]-x[i],2)+pow(y[k]-y[i],2);
    w=p[i]/d;
    if(w<minp)continue;
    (*mean)+=w;
    (*m)++;
  }
  (*mean)/=(*m);
}
