#include <stdio.h>
void main()
{
  float a1,b1,me1=0,a2,b2,me2=0,a3,b3,me3=0,mout=0,x;
  int n1=0,n2=0,n3=0,nout=0,i,n,fl;
  printf("Δώστε τα όρια του πρώτου διαστήματος (a1,b1)\n");
  scanf("%f %f",&a1,&b1);
  printf("Δώστε τα όρια του δεύτερου διαστήματος (a2,b2)\n");
  scanf("%f %f",&a2,&b2);
  printf("Δώστε τα όρια του τρίτου διαστήματος (a3,b3)\n");
  scanf("%f %f",&a3,&b3);
  printf("Δώστε το πλήθος των αριθμών n = ? ");
  scanf("%d",&n);
  printf("Δώστε %d τυχαίους αριθμούς από το πληκτρολόγιο\n",n);
  for(i=0;i<n;i++){
    scanf("%f",&x);
    fl=1;
    if(a1<x&&x<b1){
      me1+=x;
      n1++;
      fl=0;
    }
    if(a2<x&&x<b2){
      me2+=x;
      n2++;
      fl=0;
    }
    if(a3<x&&x<b3){
      me3+=x;
      n3++;
      fl=0;
    }
    if(fl){
      mout+=x;
      nout++;
    }
  }
  if(n1==0)
    printf("Στο διάστημα(%f,%f) δεν υπάρχουν τιμές\n",a1,b1);
  else{
    me1=me1/n1;
    printf("Στο διάστημα (%f,%f) βρέθηκαν %d τιμές με μέση τιμή %f\n",a1,b1,n1,me1);
  }
  if(n2==0)
    printf("Στο διάστημα(%f,%f) δεν υπάρχουν τιμές\n",a2,b2);
  else{
    me2=me2/n2;
    printf("Στο διάστημα (%f,%f) βρέθηκαν %d τιμές με μέση τιμή %f\n",a2,b2,n2,me2);
  }
  if(n3==0)
    printf("Στο διάστημα(%f,%f) δεν υπάρχουν τιμές\n",a3,b3);
  else{
    me3=me3/n3;
    printf("Στο διάστημα (%f,%f) βρέθηκαν %d τιμές με μέση τιμή %f\n",a3,b3,n3,me3);
  }
  if(nout==0)
    printf("Έξω από τα διαστήματα δεν υπάρχουν τιμές\n");
  else{
    mout/=nout;
    printf("Έξω από τα διαστήματα βρέθηκαν %d τιμές με μέση τιμή %f\n",nout,mout);
  }
}
